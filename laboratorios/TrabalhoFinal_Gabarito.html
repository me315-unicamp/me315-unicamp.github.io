import pandas as pd

resultados = []
for chunk in pd.read_csv("../dados/flights.csv.zip",
iterator=True, header = 0, chunksize=100000, usecols=['AIRLINE', 'DAY', 'MONTH', 'ARRIVAL_DELAY']):
     chunk = chunk.dropna(subset=['ARRIVAL_DELAY'])
     chunk = chunk[chunk.AIRLINE.isin(['AA', 'DL', 'UA', 'US'])]
     chunk['atraso'] = chunk['ARRIVAL_DELAY'] > 10
     resultados.append(chunk.groupby(['AIRLINE', 'MONTH', 'DAY']).atraso.agg(['sum', 'count']))
resultados = pd.concat(resultados)

resultados = resultados.groupby(['AIRLINE', 'MONTH', 'DAY']).agg({'sum': 'sum', 'count': 'sum'})
resultados['P_DELAY'] = resultados['sum']/resultados['count']
resultados = resultados.reset_index()
